2. buat user dengan /home

4. DNS server
- apt-cdrom add
- apt install bind9 dnsutils
- cd /etc/bind
- nano named.conf.local
zone "smkbisa.net" {
	type master;
	file "/etc/bind/db.domain";
};

zone "5.168.192.in-addr.arpa' {
	type master;
	file "/etc/bind/db.ip";
};

- nano named.conf.options
allow-query { any; };
forwarders {
  8.8.8.8;
};
dnssec-enable no;
dnssec-validation no;

- cp db.local db.domain
- cp db.127 db.ip
- nano db.domain
@	IN	A	smkbisa.net.
@	IN	A	192.168.5.2
internal	IN	A	192.168.5.2
info	IN	A	192.168.5.2
@	IN	MX	10	mail.smkbisa.net.
mail	IN	A	192.168.5.2
router	IN	A	182.16.5.254
ftp	IN	A	192.168.5.2
ntp	IN	A	192.168.5.2
modul	IN	A	192.168.5.2

- nano db.ip
@	IN	NS	smkbisa.net.
2	IN	PTR	smkbisa.net.

5. Web Server
- apt install apache2
- cd /etc/apache2/sites-available/
- cp 000-default.conf smkbisa.conf
- nano smkbisa.conf
ServerName smkbisa.net
- systemctl restart apache2
